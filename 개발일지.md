  const extensions = /(\.jpg|\.jpeg|\.png|\.gif|\.bmp)$/i;
  return value.length !== 0 && !extensions.exec(value) && '이미지 파일(.jpg .jpeg .png .gif .bmp)만 올려주세요.';
